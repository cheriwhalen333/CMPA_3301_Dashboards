        // Label Wrapping Logic (16 char)

        function wrap(str, max = 16) {

            if (str.length <= max) return str;

            const words = str.split(' ');

            const lines = [];

            let cur = words[0];

            for (let i = 1; i < words.length; i++) {

                if ((cur + ' ' + words[i]).length <= max) { cur += ' ' + words[i]; }

                else { lines.push(cur); cur = words[i]; }

            }

            lines.push(cur);

            return lines;

        }
